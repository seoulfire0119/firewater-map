    function showAllMarkers() {
      for (let i = 0; i < data.length; i++) {
        markers[i].setMap(map);
        markerOverlays[i].setMap(map);
      }
    };
    showAllMarkersBtn.addEventListener("click", () => {
      if (!isShowingAllMarkers) {
        showAllMarkers();
        isShowingAllMarkers = true;
      } else {
        hideAllMarkers();
        isShowingAllMarkers = false;
      }
    });
