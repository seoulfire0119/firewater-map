        function updateMarkers() {
    const checkBoxes = Array.from(document.querySelectorAll('#search-results input[type="checkbox"]'));
    checkBoxes.forEach((checkBox) => {
        const dataIndex = data.findIndex((d) => d[2] === parseInt(checkBox.id.split("-")[2], 10));

            markers[dataIndex].setMap(map);

            const hiddenMarkerIndex = hiddenMarkers.indexOf(markers[dataIndex]);
            if (hiddenMarkerIndex === -1) {
                hiddenMarkers.push(markers[dataIndex]);
            }
        }
    });

  function getMarkerImage(color) {
    return new kakao.maps.MarkerImage(
      color === "red" ? redMarkerImageSrc : blueMarkerImageSrc,
      new kakao.maps.Size(24, 35)
    );
  }

  function initializeMarkers() {
    data.forEach((coordinates, index) => {
        var marker = new kakao.maps.Marker({
            position: new kakao.maps.LatLng(coordinates[0], coordinates[1]),
        });

        marker.completed = false;

        markers.push(marker);

        const markerOverlay = new kakao.maps.CustomOverlay({
            content: `<div style="font-size: 30px; color: #000; padding: 0; background-color: rgba(255, 255, 255, 0.7);">${coordinates[2]}</div>`,
            position: marker.getPosition(),
            yAnchor: 3
        });
        markerOverlay.setMap(map);

        kakao.maps.event.addListener(marker, "click", function () {
            var content = `<div style="padding:5px;">일련번호: ${data[index][2]}<br><button onclick="toggleMarkerStatus(${index});">${
                marker.completed ? "용수확인 취소" : "용수확인 완료"
            }</button></div>`;

            var infowindow = new kakao.maps.InfoWindow({
                content: content,
            });

            if (currInfowindow) {
                currInfowindow.close();
            }
            currInfowindow = infowindow;

            infowindow.open(map, marker);
        });
    });
}


  function toggleMarkerStatus(index) {
    var marker = markers[index];

    marker.completed = !marker.completed;
    marker.setImage(getMarkerImage(marker.completed ? "red" : "blue"));

    currInfowindow.close();
  }

  function initializeCurrentPositionMarker() {
          if (navigator.geolocation) {
              navigator.geolocation.getCurrentPosition(function (position) {
                  const currentPosition = new kakao.maps.LatLng(position.coords.latitude, position.coords.longitude);
                  const currentPositionMarker = new kakao.maps.Marker({
                      position: currentPosition,
                      image: new kakao.maps.MarkerImage(
                          'https://cdn.pixabay.com/photo/2016/03/31/20/56/map-marker-1295839_1280.png',
                          new kakao.maps.Size(24, 24),
                          {
                              offset: new kakao.maps.Point(12, 12),
                          }
                      ),
                  });
                  currentPositionMarker.setMap(map);
                  watchCurrentPosition(currentPositionMarker);
              });
          } else {
              alert('이 브라우저에서 지오로케이션을 사용할 수 없습니다.');
          }
      }

      // 사용자의 현재 위치를 지속적으로 파악하고 전달받은 마커의 위치를 변경시키는 함수입니다.
      function watchCurrentPosition(marker) {
          if (navigator.geolocation) {
              navigator.geolocation.watchPosition(function (position) {
                  const currentPosition = new kakao.maps.LatLng(position.coords.latitude, position.coords.longitude);
                  marker.setPosition(currentPosition);
              });
          } else {
              alert('이 브라우저에서 지오로케이션을 사용할 수 없습니다.');
          }
      }

      // '내 위치 찾기' 버튼을 클릭하면 지도의 중심을 사용자의 현재 위치로 이동시키는 함수입니다.
      function moveToCurrentLocation() {
          if (navigator.geolocation) {
              navigator.geolocation.getCurrentPosition(function (position) {
                  var currentPosition = new kakao.maps.LatLng(position.coords.latitude, position.coords.longitude);
                  map.setCenter(currentPosition);
              });
          } else {
              alert('이 브라우저에서 지오로케이션을 사용할 수 없습니다.');
          }
      }

      // 현재 위치 마커를 초기화하고 지도에 표시합니다.
