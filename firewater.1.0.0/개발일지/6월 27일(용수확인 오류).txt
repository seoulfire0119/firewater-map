// 체크박스 활성화에 따른 마커 업데이트
function updateMarkers() {
    const checkBoxes = Array.from(document.querySelectorAll('#search-results input[type="checkbox"]'));

    checkBoxes.forEach((checkBox) => {
        const dataIndex = data.findIndex((d) => d[2] === parseInt(checkBox.id.split("-")[2], 10));
        const isChecked = checkBox.checked;

        // 사용자가 선택한 지역 정보가 있고, 해당 데이터가 선택한 지역 정보와 일치하지 않으면 다음 데이터를 처리합니다.
        if (selectedGu && data[dataIndex][3] !== selectedGu) {
            return;
        }

        // 체크박스 체크가 되어있다면 맵에 표시됨
        if (isChecked) {
            markers[dataIndex].setMap(map);
            markerOverlays[dataIndex].setMap(map);
            map.panTo(markers[dataIndex].getPosition());
        // 체크박스가 체크 되어있지 않으면 맵에서 표시안됨
        } else {
            markers[dataIndex].setMap(null);
            markerOverlays[dataIndex].setMap(null);
        }
    });
}

function updateSearchResults() {
    const searchValue = document.getElementById("search").value;
    const guValue = document.getElementById("gu-select").value;
    const centerValue = document.getElementById("center-select").value;

    const searchResults = data.filter((d) => {
        const isSearchMatch = String(d[2]).startsWith(searchValue);
        const isGuMatch = d[3] === guValue || !guValue;
        const isCenterMatch = d[4] === centerValue || !centerValue;
        const isChecked = checkedBoxes[d[2]] || false;

        return isChecked || (isSearchMatch && isGuMatch && isCenterMatch);
    });

    

     // 추가된 부분: 정렬을 위한 코드

        const existingResult = document.getElementById(`search-result-${data[dataIndex][2]}`);
        if (existingResult) return;

        const div = document.createElement("div");
        div.className = "search-result";
        div.innerHTML = `
    <input type="checkbox" id="search-result-${data[dataIndex][2]}" ${checkedBoxes[data[dataIndex][2]] ? 'checked' : ''}>
    <label for="search-result-${data[dataIndex][2]}">${data[dataIndex][2]}</label>
    `;
        searchResultsDiv.appendChild(div);

        const checkbox = document.querySelector(`#search-result-${data[dataIndex][2]}`);

        checkbox.addEventListener("change", (event) => {
            checkedBoxes[data[dataIndex][2]] = event.target.checked;
            updateMarkers();
        });
    });
}
